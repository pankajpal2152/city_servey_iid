## 4. Test Database Connectivity

This test connects to RDS, confirms the active database, confirms the user and
customer stored procedures exist, and runs the sample authentication procedure.

```powershell
@'
require('dotenv').config();
const mysql = require('mysql2/promise');

(async () => {
  const connection = await mysql.createConnection({
    host: process.env.DB_HOST,
    user: process.env.DB_USER,
    password: process.env.DB_PASS,
    database: process.env.DB_NAME,
    multipleStatements: true,
  });

  const [[dbInfo]] = await connection.query('SELECT DATABASE() AS database_name, 1 AS connection_ok');
  console.log('DB connection:', dbInfo);

  const [procedures] = await connection.query(
    `SELECT ROUTINE_NAME
       FROM INFORMATION_SCHEMA.ROUTINES
      WHERE ROUTINE_SCHEMA = ?
        AND ROUTINE_TYPE = 'PROCEDURE'
        AND ROUTINE_NAME IN (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
      ORDER BY ROUTINE_NAME`,
    [
      process.env.DB_NAME,
      'USP_POST_USER_AUTHENTICATE_ACTIVITY',
      'USP_GET_USER_LIST_ACTIVITY',
      'USP_POST_USER_PROFILE_ACTIVITY',
      'USP_GET_LIST_CUSTOMER_ACTIVITY',
      'USP_GET_SPECIFIC_CUSTOMER_ACTIVITY',
      'USP_POST_CUSTOMER_DETAILS_ACTIVITY',
      'USP_GET_LIST_PROJECT_ACTIVITY',
      'USP_POST_PROJECT_ACTIVITY',
      'USP_GET_ALL_MASTER_DATA',
      'USP_POST_ALL_MASTER_DATA',
    ]
  );
  console.log('Procedures found:', procedures.map((row) => row.ROUTINE_NAME));

  const [authRows] = await connection.execute(
    'CALL USP_POST_USER_AUTHENTICATE_ACTIVITY(?, ?, @ERRNO, @ERRMSG);',
    ['AUTHENTICATE_USER', JSON.stringify([{ USER_NAME: 'sai@yopmail.com', PASSWORD: 'Abc@1234' }])]
  );
  const firstResult = authRows?.[0]?.[0]?.JSON_VALUE;
  console.log('Authenticate procedure returned:', typeof firstResult === 'string' ? firstResult : JSON.stringify(firstResult));

  await connection.end();
})().catch((error) => {
  console.error('DB test failed:', error.message);
  process.exit(1);
});
'@ | node -
```

Expected:

- `database_name` is `DEV_CITY_SURVEY`
- All required procedure names are printed
- Auth procedure returns `status: true`

## 8. Automated Local Smoke Script

This script starts the app on a temporary port, logs in, calls the protected
endpoint with the existing sample user, and verifies it does not create a
duplicate user.

```powershell
@'
require('dotenv').config();
const { spawn } = require('child_process');
const mysql = require('mysql2/promise');

const port = '3103';
const baseUrl = `http://127.0.0.1:${port}`;
const sampleUser = {
  ITEM: 'ADD_USER',
  USER_SYS_ID: 0,
  SYSTEM_ROLE_SYS_ID: 1,
  FIRST_NAME: 'Sai',
  LAST_NAME: 'Roy',
  GENDER: 'Male',
  MOBILE_NO: '7894561230',
  EMAIL_ID: 'sai@yopmail.com',
  ORGANISATION_SYS_ID: '1',
  PASSWORD: 'Abc@1234',
  CREATED_BY: '1',
};

async function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

async function waitForServer() {
  for (let attempt = 0; attempt < 40; attempt += 1) {
    try {
      const response = await fetch(`${baseUrl}/health`);
      if (response.status === 200) return;
    } catch (error) {
      await sleep(500);
    }
  }
  throw new Error('Server did not become ready in time');
}

async function getEmailCount(connection) {
  const [[row]] = await connection.query(
    'SELECT COUNT(*) AS count FROM TBL_USER_DETAILS WHERE EMAIL_ID = ?',
    [sampleUser.EMAIL_ID]
  );
  return Number(row.count);
}

(async () => {
  const server = spawn(process.execPath, ['index.js'], {
    cwd: process.cwd(),
    env: { ...process.env, PORT: port },
    stdio: ['ignore', 'pipe', 'pipe'],
  });

  const connection = await mysql.createConnection({
    host: process.env.DB_HOST,
    user: process.env.DB_USER,
    password: process.env.DB_PASS,
    database: process.env.DB_NAME,
  });

  try {
    await waitForServer();

    const healthResponse = await fetch(`${baseUrl}/health`);
    const healthJson = await healthResponse.json();

    const authResponse = await fetch(`${baseUrl}/api/auth/api-post-authenticate-user`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify([{ USER_NAME: sampleUser.EMAIL_ID, PASSWORD: sampleUser.PASSWORD }]),
    });
    const authJson = await authResponse.json();

    const userListResponse = await fetch(`${baseUrl}/api/user/api-get-view-user-list-info?ITEM=VIEW_ALL`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const userListJson = await userListResponse.json();

    const beforeCount = await getEmailCount(connection);
    const userResponse = await fetch(`${baseUrl}/api/user/api-post-add-update-user`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify(sampleUser),
    });
    const userJson = await userResponse.json();
    const afterCount = await getEmailCount(connection);

    const customerValidationResponse = await fetch(`${baseUrl}/api/customer/api-post-add-update-customer-details`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify({ ITEM: 'ADD' }),
    });
    const customerValidationJson = await customerValidationResponse.json();

    const customerListResponse = await fetch(`${baseUrl}/api/customer/api-get-view-list-customer-details?ITEM=VIEW_ALL`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const customerListJson = await customerListResponse.json();

    const specificCustomerResponse = await fetch(`${baseUrl}/api/customer/api-get-view-specific-customer-details?ITEM=PROFILE&CUSTOMER_SYS_ID=4`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const specificCustomerJson = await specificCustomerResponse.json();

    const projectListResponse = await fetch(`${baseUrl}/api/project/api-get-view-list-project-details?ITEM=VIEW_ALL`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const projectListJson = await projectListResponse.json();

    const systemRoleListResponse = await fetch(`${baseUrl}/api/master/api-get-view-master-system-roles?ITEM=VIEW_ALL&RECORD_SYS_ID=0&ORGANIZATION_SYS_ID=0`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const systemRoleListJson = await systemRoleListResponse.json();

    const surveyTypeListResponse = await fetch(`${baseUrl}/api/master/api-get-view-master-survey-type?ITEM=VIEW_ALL&RECORD_SYS_ID=0&ORGANIZATION_SYS_ID=0`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const surveyTypeListJson = await surveyTypeListResponse.json();

    const propertyTypeListResponse = await fetch(`${baseUrl}/api/master/api-get-view-master-property-type?ITEM=VIEW_ALL&RECORD_SYS_ID=0&ORGANIZATION_SYS_ID=0`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${authJson.Token}`,
      },
    });
    const propertyTypeListJson = await propertyTypeListResponse.json();

    const projectValidationResponse = await fetch(`${baseUrl}/api/project/api-post-add-update-project`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify({ ITEM: 'ADD' }),
    });
    const projectValidationJson = await projectValidationResponse.json();

    const systemRoleValidationResponse = await fetch(`${baseUrl}/api/master/api-post-add-update-master-system-role`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify({ ITEM: 'ADD' }),
    });
    const systemRoleValidationJson = await systemRoleValidationResponse.json();

    const surveyTypeValidationResponse = await fetch(`${baseUrl}/api/master/api-post-add-update-master-survey-type`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify({ ITEM: 'ADD' }),
    });
    const surveyTypeValidationJson = await surveyTypeValidationResponse.json();

    const propertyTypeValidationResponse = await fetch(`${baseUrl}/api/master/api-post-add-update-master-property-type`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${authJson.Token}`,
      },
      body: JSON.stringify({ ITEM: 'ADD' }),
    });
    const propertyTypeValidationJson = await propertyTypeValidationResponse.json();

    console.log({
      health_http_status: healthResponse.status,
      health_status: healthJson.status,
      auth_http_status: authResponse.status,
      auth_status: authJson.status,
      token_returned: Boolean(authJson.Token),
      user_list_http_status: userListResponse.status,
      user_list_status: userListJson.status,
      add_update_http_status: userResponse.status,
      add_update_response_status: userJson.status,
      add_update_response: userJson.response,
      sample_email_count_before: beforeCount,
      sample_email_count_after: afterCount,
      created_extra_sample_user: afterCount > beforeCount,
      customer_validation_http_status: customerValidationResponse.status,
      customer_validation_response: customerValidationJson.response,
      customer_list_http_status: customerListResponse.status,
      customer_list_status: customerListJson.status,
      specific_customer_http_status: specificCustomerResponse.status,
      specific_customer_status: specificCustomerJson.status,
      project_list_http_status: projectListResponse.status,
      project_list_status: projectListJson.status,
      project_validation_http_status: projectValidationResponse.status,
      project_validation_response: projectValidationJson.response,
      system_role_list_http_status: systemRoleListResponse.status,
      system_role_list_status: systemRoleListJson.status,
      system_role_validation_http_status: systemRoleValidationResponse.status,
      system_role_validation_response: systemRoleValidationJson.response,
      survey_type_list_http_status: surveyTypeListResponse.status,
      survey_type_list_status: surveyTypeListJson.status,
      survey_type_validation_http_status: surveyTypeValidationResponse.status,
      survey_type_validation_response: surveyTypeValidationJson.response,
      property_type_list_http_status: propertyTypeListResponse.status,
      property_type_list_status: propertyTypeListJson.status,
      property_type_validation_http_status: propertyTypeValidationResponse.status,
      property_type_validation_response: propertyTypeValidationJson.response,
    });
  } finally {
    await connection.end();
    server.kill();
    await sleep(500);
  }
})().catch((error) => {
  console.error('Smoke test failed:', error.message);
  process.exit(1);
```

## 9. Write Test Warning

To test actual user creation, change `EMAIL_ID` and `MOBILE_NO` to unique values.

Only do this when the team agrees to create a real record in the dev database,
because the stored procedure commits internally.

To test actual customer creation, call
`POST /api/customer/api-post-add-update-customer-details` with the full customer
and `PIC` sample. Only do this when the team agrees to create real customer and
PIC records in the dev database.

To test actual project creation, call
`POST /api/project/api-post-add-update-project` with the full project sample.
Only do this when the team agrees to create real project records in the dev
database.

To test actual system role creation, call
`POST /api/master/api-post-add-update-master-system-role` with the full system
role sample. Only do this when the team agrees to create real system role
records in the dev database.

To test actual survey type creation, call
`POST /api/master/api-post-add-update-master-survey-type` with the full survey
type sample. Only do this when the team agrees to create real survey type
records in the dev database.

To test actual property type creation, call
`POST /api/master/api-post-add-update-master-property-type` with the full
property type sample. Only do this when the team agrees to create real property
type records in the dev database.
